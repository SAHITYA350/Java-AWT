//Login page using java awt

import java.awt.*;
import java.awt.event.*;
import javax.swing.*;

public class Main extends Frame {
    private Label lblUsername, lblPassword;
    private TextField txtUsername, txtPassword;
    private Button btnLogin, btnReset;
    private Panel panel;

    public Main() {
        super("Login Page");
        setBackground(Color.CYAN);
        setLayout(new FlowLayout());

        panel = new Panel();
        panel.setLayout(new GridLayout(3, 2));
        panel.setBackground(Color.LIGHT_GRAY);

        lblUsername = new Label("Username:");
        lblPassword = new Label("Password:");
        txtUsername = new TextField(20);
        txtPassword = new TextField(20);
        txtPassword.setEchoChar('*');
        btnLogin = new Button("Login");
        btnReset = new Button("Reset");

        panel.add(lblUsername);
        panel.add(txtUsername);
        panel.add(lblPassword);
        panel.add(txtPassword);
        panel.add(btnLogin);
        panel.add(btnReset);

        add(panel);

        // Action listeners for buttons
        btnLogin.addActionListener(new ActionListener() {
            public void actionPerformed(ActionEvent e) {
                String username = txtUsername.getText();
                String password = txtPassword.getText();
                // Add authentication logic here
                if (username.equals("admin") && password.equals("password")) {
                    JOptionPane.showMessageDialog(null, "Login successful!");
                    System.exit(0);
                } else {
                    JOptionPane.showMessageDialog(null, "Invalid username or password");
                }
            }
        });

        btnReset.addActionListener(new ActionListener() {
            public void actionPerformed(ActionEvent e) {
                txtUsername.setText("");
                txtPassword.setText("");
            }
        });

        setSize(300, 200);
        setVisible(true);
    }

    public static void main(String[] args) {
        new Main();
    }
}

//Press button and background colour changed

import java.awt.*;
import java.awt.event.*;

class Test implements ActionListener {
    private Frame f;
    private Button b1, b2;
    private Panel p;


    Test() {
        f = new Frame("My coloure:");
        p = new Panel();
        b1 = new Button("Red");
        b2 = new Button("Blue");
        f.add(p);
        p.add(b1);
        p.add(b2);
        b1.setBackground(Color.red);
        b2.setBackground(Color.blue);
        b1.addActionListener(this);
        b2.addActionListener(this);
        f.setBackground(Color.black);
        f.setSize(400, 400);
        f.setVisible(true);
    }

    public void actionPerformed(ActionEvent e) {
        Color color = null;

        if (e.getSource() == b1) {
            color = Color.RED;
            System.out.println("Red working.");
        } else if (e.getSource() == b2) {
            color = Color.BLUE;
            System.out.println("Blue working.");
        }

        if (color != null) {
            f.setBackground(color);
            p.setBackground(color);
        }
    }

    public static void main(String[] args) {
       Test t1= new Test();
    }
}
